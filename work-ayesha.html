KMP Algorithm - Library Search

#include <iostream>
#include <vector>
#include <string>
using namespace std;

// Function to compute the LPS array
void computeLPSArray(const string& pattern, vector<int>& lps) {
    int m = pattern.length();
    int len = 0;
    lps[0] = 0;
    int i = 1;

    while (i < m) {
        if (pattern[i] == pattern[len]) {
            len++;
            lps[i] = len;
            i++;
        } else {
            if (len != 0) {
                len = lps[len - 1];
            } else {
                lps[i] = 0;
                i++;
            }
        }
    }
}

// Function to perform KMP search
vector<string> KMPSearch(const vector<string>& bookTitles, const string& query) {
    vector<string> results;
    for (const string& title : bookTitles) {
        int n = title.length();
        int m = query.length();
        vector<int> lps(m);
        computeLPSArray(query, lps);

        int i = 0, j = 0; // Pointers for title and query
        while (i < n) {
            if (query[j] == title[i]) {
                i++;
                j++;
            }
            if (j == m) {
                results.push_back(title);
                j = lps[j - 1];
            } else if (i < n && query[j] != title[i]) {
                if (j != 0) {
                    j = lps[j - 1];
                } else {
                    i++;
                }
            }
        }
    }
    return results;
}

// Example usage
int main() {
    vector<string> bookTitles = {
        "Introduction to Algorithms",
        "Design Patterns",
        "Data Structures and Algorithms",
        "Advanced C++ Programming",
        "Machine Learning Basics"
    };
    string query = "Algorithms";

    vector<string> result = KMPSearch(bookTitles, query);
    cout << "Books matching your query:" << endl;
    for (const string& title : result) {
        cout << "- " << title << endl;
    }

    return 0;
}

BST Algorithm - Course Registration

#include <iostream>
using namespace std;

struct Node {
    int courseID;
    string courseName;
    Node* left;
    Node* right;

    Node(int id, string name) : courseID(id), courseName(name), left(nullptr), right(nullptr) {}
};

// Insert a course into the BST
Node* insert(Node* root, int courseID, string courseName) {
    if (!root) return new Node(courseID, courseName);
    if (courseID < root->courseID) root->left = insert(root->left, courseID, courseName);
    else root->right = insert(root->right, courseID, courseName);
    return root;
}

// Search for a course
Node* search(Node* root, int courseID) {
    if (!root || root->courseID == courseID) return root;
    if (courseID < root->courseID) return search(root->left, courseID);
    return search(root->right, courseID);
}

// In-order traversal
void inOrder(Node* root) {
    if (root) {
        inOrder(root->left);
        cout << root->courseID << ": " << root->courseName << endl;
        inOrder(root->right);
    }
}

int main() {
    Node* root = nullptr;
    root = insert(root, 101, "Data Structures");
    root = insert(root, 102, "Algorithms");
    root = insert(root, 103, "Operating Systems");

    cout << "Course List:" << endl;
    inOrder(root);

    int searchID = 102;
    Node* result = search(root, searchID);
    if (result) cout << "Course Found: " << result->courseName << endl;
    else cout << "Course Not Found." << endl;

    return 0;
}

Fenwick Tree - Grade Tracking

#include <iostream>
#include <vector>
using namespace std;

class FenwickTree {
    vector<int> tree;
    int n;

public:
    FenwickTree(int size) : n(size) {
        tree.resize(n + 1, 0);
    }

    void update(int index, int value) {
        while (index <= n) {
            tree[index] += value;
            index += index & -index;
        }
    }

    int query(int index) {
        int sum = 0;
        while (index > 0) {
            sum += tree[index];
            index -= index & -index;
        }
        return sum;
    }
};

int main() {
    FenwickTree ft(10);

    ft.update(1, 90); // Add marks for student 1
    ft.update(2, 85); // Add marks for student 2
    ft.update(3, 75); // Add marks for student 3

    cout << "Total marks for student 1 to 3: " << ft.query(3) << endl;

    return 0;
}
